import nexus

# Load the Kaggle Bank Customer Churn model
model = nexus.models.load("Bank-Churn-XGBoost:v2.4.1")

# Deploy to production cluster with drift guard enabled
deploy = model.deploy(
    target="prod-cluster-us-east",
    drift_detection=True,
    drift_threshold=0.20
)
print(f"Deployment online: {deploy.endpoint}")
